### `crates/voss-sdk/src/error.rs` (utility — no analog)

**No direct analog** — `voss-tui` uses `anyhow` throughout; `thiserror` is in the workspace but not used for a public error type in any analog crate. The pattern is standard Rust library design.

**Error enum shape (from RESEARCH.md pattern + `ok_or_detail` in net.rs lines 24–37)**:
```rust
use thiserror::Error;

/// All errors returned by the voss-sdk public API.
/// Must NOT contain the bearer token in any Display or Debug output.
#[derive(Debug, Error)]
pub enum VossError {
    /// HTTP request failed at the transport level.
    #[error("HTTP request error: {0}")]
    Http(#[from] reqwest::Error),

    /// Server returned a non-2xx status with a detail message.
    /// Maps from net.rs's `ok_or_detail` pattern (lines 24–37).
    #[error("HTTP {status}: {detail}")]
    HttpStatus { status: u16, detail: String },

    /// SSE stream framing error.
    #[error("SSE error: {0}")]
    Sse(String),

    /// Event deserialization failed.
    #[error("decode error: {0}")]
    Decode(String),

    /// Server handshake failed or timed out.
    #[error("handshake error: {0}")]
    Handshake(String),

    /// Supervisor spawn failed (bad interpreter path, etc).
    #[error("spawn error: {0}")]
    Spawn(#[from] std::io::Error),
}
```

**Library boundary rule (RESEARCH.md pitfall 2):** All `pub fn` return types use `Result<T, VossError>`. Private helpers may use `anyhow` internally but must convert with `.map_err(...)` at the module boundary. `anyhow` must NOT appear in any public signature.

```rust
async fn ok_or_detail(resp: reqwest::Response) -> Result<reqwest::Response, VossError> {
    let status = resp.status();
    if status.is_success() {
        return Ok(resp);
    }
    let body = resp.text().await.unwrap_or_default();
    let detail = serde_json::from_str::<serde_json::Value>(&body)
        .ok()
        .and_then(|v| v.get("detail").and_then(|d| d.as_str()).map(String::from));
    Err(VossError::HttpStatus {
        status: status.as_u16(),
        detail: detail.unwrap_or_else(|| format!("HTTP {status}")),
    })
}
```

```rust
pub async fn spawn_with(python: &str, extra_env: &[(&str, &str)]) -> Result<Supervisor, VossError> {
    let mut cmd = Command::new(python);
    cmd.args(["-m", "voss.cli", "serve", "--port", "0"])
        .stdin(Stdio::piped())   // held open = heartbeat
        .stdout(Stdio::piped())
        .stderr(Stdio::inherit())
        .kill_on_drop(true);     // belt-and-suspenders (Drop impl is the suspenders)
    for (k, v) in extra_env {
        cmd.env(k, v);
    }
    let mut child = cmd.spawn()
        .map_err(VossError::Spawn)?;  // was: cmd.spawn()?  with anyhow

    let stdout = child.stdout.take()
        .ok_or_else(|| VossError::Handshake("no stdout pipe".into()))?;
    let mut lines = BufReader::new(stdout).lines();

    let hs: Handshake = tokio::time::timeout(
        std::time::Duration::from_secs(20),
        async {
            while let Some(line) = lines.next_line().await
                .map_err(|e| VossError::Handshake(e.to_string()))?
            {
                if let Ok(h) = serde_json::from_str::<Handshake>(&line) {
                    return Ok(h);
                }
            }
            Err(VossError::Handshake("server exited before handshake".into()))
        },
    )
    .await
    .map_err(|_| VossError::Handshake("server handshake timed out".into()))??;

    // Drain remaining stdout (same as server.rs line 86)
    tokio::spawn(async move { while let Ok(Some(_)) = lines.next_line().await {} });

    let base = format!("http://127.0.0.1:{}", hs.port);
    let client = VossClient::new(base, hs.token);
    Ok(Supervisor { child, client })
}
```

```rust
async fn ok_or_detail(resp: reqwest::Response) -> Result<reqwest::Response, VossError> {
    let status = resp.status();
    if status.is_success() { return Ok(resp); }
    let body = resp.text().await.unwrap_or_default();
    let detail = serde_json::from_str::<serde_json::Value>(&body)
        .ok()
        .and_then(|v| v.get("detail").and_then(|d| d.as_str()).map(String::from));
    Err(VossError::HttpStatus {
        status: status.as_u16(),
        detail: detail.unwrap_or_else(|| format!("HTTP {status}")),
    })
}
```
