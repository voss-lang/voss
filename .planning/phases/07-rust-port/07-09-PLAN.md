    ```python
    """Persisted dispatcher decision per D-04 + D-06."""
    from __future__ import annotations
    import os
    from pathlib import Path

    try:
        import tomllib  # py311+
    except ImportError:
        import tomli as tomllib  # type: ignore

    try:
        import tomli_w  # for writing
    except ImportError:
        tomli_w = None

    CONFIG_KEY = "dispatcher.allow_autodownload"

    def config_path() -> Path:
        base = Path(os.environ.get("XDG_CONFIG_HOME", Path.home() / ".config"))
        return base / "voss" / "config.toml"

    def load() -> dict:
        p = config_path()
        if not p.exists():
            return {}
        try:
            return tomllib.loads(p.read_text())
        except Exception:
            return {}

    def get_allow_autodownload() -> bool | None:
        d = load()
        # Nested key: dispatcher.allow_autodownload
        return d.get("dispatcher", {}).get("allow_autodownload")

    def set_allow_autodownload(value: bool) -> None:
        p = config_path()
        p.parent.mkdir(parents=True, exist_ok=True)
        d = load()
        d.setdefault("dispatcher", {})["allow_autodownload"] = value
        if tomli_w is None:
            # Minimal fallback: write a flat-style TOML by hand.
            text = "[dispatcher]\n"
            text += f"allow_autodownload = {'true' if value else 'false'}\n"
            p.write_text(text)
        else:
            p.write_bytes(tomli_w.dumps(d).encode("utf-8"))
    ```

    ```python
    """Auto-download + exec the Rust voss-cli binary on first agent-verb invocation.

    Implements D-04..D-07 from .planning/phases/07-rust-port/07-CONTEXT.md.

    - D-07: compiler verbs (compile, run, check, init, ast) NEVER auto-download.
    - AGENT_VERBS is exactly the v1.2-scoped agent surface per 07-PHASE.md +
      07-CONTEXT.md D-04..D-07. `tools` and `config` are NOT v1.2 scope and
      MUST NOT appear here.
    """
    from __future__ import annotations
    import hashlib
    import os
    import platform
    import shutil
    import sys
    import urllib.request
    from pathlib import Path

    from . import _dispatcher_config as cfg

    # Exactly five agent verbs in v1.2 scope (per 07-PHASE.md + 07-CONTEXT.md).
    # Do NOT add `tools` or `config` here — they are not v1.2 scope.
    # `--version` is special-cased upstream (in voss/cli.py) and never reaches
    # this set, so it never triggers download (it is the Phase 7 perf gate).
    # `/save` is a slash-command inside the chat REPL, not a verb.
    AGENT_VERBS = {"do", "chat", "doctor", "sessions", "resume"}
    COMPILER_VERBS = {"compile", "run", "check", "init", "ast"}  # D-07: never dispatch

    GITHUB_REPO = os.environ.get("VOSS_GITHUB_REPO", "voss-lang/voss")

    def _target_triple() -> str | None:
        sys_name = platform.system()
        machine = platform.machine().lower()
        if sys_name == "Darwin":
            if machine in ("arm64", "aarch64"): return "aarch64-apple-darwin"
            if machine in ("x86_64", "amd64"): return "x86_64-apple-darwin"
        if sys_name == "Linux":
            if machine in ("aarch64", "arm64"): return "aarch64-unknown-linux-gnu"
            if machine in ("x86_64", "amd64"): return "x86_64-unknown-linux-gnu"
        return None

    def _local_bin_path() -> Path:
        return Path.home() / ".local" / "bin" / "voss-cli"

    def _find_voss_cli() -> str | None:
        """Resolve voss-cli on PATH or in ~/.local/bin."""
        path = shutil.which("voss-cli")
        if path: return path
        local = _local_bin_path()
        return str(local) if local.is_file() and os.access(local, os.X_OK) else None

    def _prompt_yn(message: str) -> bool:
        if not sys.stdin.isatty():
            return False
        sys.stderr.write(message)
        sys.stderr.write(" [y/N] ")
        sys.stderr.flush()
        line = sys.stdin.readline().strip().lower()
        return line in ("y", "yes")

    def _download_and_verify(triple: str) -> Path:
        """Fetch the binary tarball, verify SHA256 against the release manifest, extract, return path."""
        import tarfile, tempfile, json
        # Latest release.
        api_url = f"https://api.github.com/repos/{GITHUB_REPO}/releases/latest"
        with urllib.request.urlopen(api_url, timeout=30) as resp:
            release = json.loads(resp.read())
        tag = release["tag_name"]
        asset_name = f"voss-cli-{triple}.tar.xz"
        sha_name = f"voss-cli-{triple}.tar.xz.sha256"
        url = f"https://github.com/{GITHUB_REPO}/releases/download/{tag}/{asset_name}"
        sha_url = f"https://github.com/{GITHUB_REPO}/releases/download/{tag}/{sha_name}"

        # Fetch SHA256 expectation.
        with urllib.request.urlopen(sha_url, timeout=30) as resp:
            expected = resp.read().decode().split()[0].strip()

        # Fetch tarball.
        tmp = Path(tempfile.mkdtemp(prefix="voss-dl-"))
        tarball = tmp / asset_name
        with urllib.request.urlopen(url, timeout=120) as resp, open(tarball, "wb") as f:
            shutil.copyfileobj(resp, f)

        # Verify.
        h = hashlib.sha256()
        with open(tarball, "rb") as f:
            for chunk in iter(lambda: f.read(8192), b""): h.update(chunk)
        actual = h.hexdigest()
        if actual != expected:
            raise RuntimeError(f"SHA256 mismatch: expected {expected}, got {actual}")

        # Extract.
        with tarfile.open(tarball, "r:xz") as tf:
            tf.extractall(tmp)
        # Find the binary.
        for p in tmp.rglob("voss-cli"):
            if p.is_file():
                dest = _local_bin_path()
                dest.parent.mkdir(parents=True, exist_ok=True)
                shutil.copy2(p, dest)
                dest.chmod(0o755)
                return dest
        raise RuntimeError("voss-cli not found in extracted tarball")

    def maybe_dispatch(argv: list[str]) -> None:
        """If argv[0] is an agent verb and voss-cli is missing, prompt + download + exec.

        argv: positional args AFTER the `voss` script name; argv[0] is the subcommand.

        Returns normally if dispatch was not triggered (compiler verbs, or user declined).
        Raises SystemExit on `os.execvp` (which never returns) or on hard-fail exits.
        """
        if not argv: return
        verb = argv[0]
        # D-07: compiler verbs never trigger dispatch.
        # Anything not in AGENT_VERBS (including `tools`, `config`, `--version`,
        # and any future Click subcommand not yet in v1.2 scope) also never triggers.
        if verb in COMPILER_VERBS or verb not in AGENT_VERBS:
            return
        # If a Rust binary is already reachable, exec it directly (no prompt).
        existing = _find_voss_cli()
        if existing:
            os.execvp(existing, [existing, *argv])
            return  # unreachable

        # Decide whether to download.
        decision = cfg.get_allow_autodownload()
        if decision is False:
            sys.stderr.write(
                "voss-cli (Rust binary) not installed and auto-download is disabled.\n"
                "Install via:  brew install voss-lang/tap/voss\n"
                "         or:  curl -sSL https://github.com/voss-lang/voss/releases/latest/download/install.sh | sh\n"
                "Re-enable auto-download: rm ~/.config/voss/config.toml\n"
            )
            sys.exit(2)
        if decision is None:
            allow = _prompt_yn(
                "voss-cli (Rust binary) is not installed.\n"
                "Download the matching release and install to ~/.local/bin/voss-cli?")
            cfg.set_allow_autodownload(allow)
            if not allow:
                sys.stderr.write(
                    "Auto-download declined. Install manually:\n"
                    "  brew install voss-lang/tap/voss\n")
                sys.exit(2)

        triple = _target_triple()
        if triple is None:
            sys.stderr.write(f"unsupported platform: {platform.system()}/{platform.machine()}\n")
            sys.exit(2)
        try:
            binary = _download_and_verify(triple)
        except Exception as e:
            sys.stderr.write(f"voss-cli download failed: {e}\n")
            sys.exit(2)
        os.execvp(str(binary), [str(binary), *argv])
    ```
