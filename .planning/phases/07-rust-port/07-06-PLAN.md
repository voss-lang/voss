    use std::path::{Path, PathBuf};
    use serde::{Deserialize, Serialize};
    use chrono::{DateTime, Utc};

    fn state_dir() -> PathBuf {
        std::env::var("XDG_STATE_HOME").map(PathBuf::from)
            .unwrap_or_else(|_| dirs::home_dir().unwrap_or_default().join(".local").join("state"))
            .join("voss").join("sessions")
    }

    pub fn session_path(id: &str) -> PathBuf { state_dir().join(format!("{}.json", id)) }

    /// Mirrors voss/harness/session.py:SessionRecord field-for-field.
    /// Order in JSON: id, name, cwd, model, started_at, updated_at, total_cost_usd, turns.
    #[derive(Clone, Debug, Serialize, Deserialize)]
    pub struct SessionRecord {
        pub id: String,
        pub name: String,
        pub cwd: String,
        pub model: String,
        pub started_at: String,    // ISO-8601 UTC, seconds precision (matches Python isoformat(timespec="seconds"))
        pub updated_at: String,
        #[serde(default)]
        pub total_cost_usd: f64,
        #[serde(default)]
        pub turns: Vec<Turn>,
    }

    /// Each turn matches the Python EpisodicMemory entry shape: at minimum `role` and `content`.
    /// Extra fields (timestamps, etc.) preserved via flatten.
    #[derive(Clone, Debug, Serialize, Deserialize)]
    pub struct Turn {
        pub role: String,
        pub content: String,
        #[serde(flatten)]
        pub extra: std::collections::BTreeMap<String, serde_json::Value>,
    }

    impl SessionRecord {
        pub fn new(cwd: &Path, model: &str, name: Option<&str>) -> Self {
            let id = uuid::Uuid::new_v4().simple().to_string()[..12].to_string();
            let now = iso_now();
            Self {
                id: id.clone(),
                name: name.map(String::from).unwrap_or_else(|| format!("session-{}", &id[..8])),
                cwd: cwd.canonicalize().unwrap_or_else(|_| cwd.to_path_buf()).to_string_lossy().into_owned(),
                model: model.into(),
                started_at: now.clone(),
                updated_at: now,
                total_cost_usd: 0.0,
                turns: Vec::new(),
            }
        }

        pub fn first_task(&self) -> String {
            for t in &self.turns {
                if t.role == "user" {
                    return t.content.chars().take(60).collect();
                }
            }
            "(empty)".into()
        }
    }

    pub fn iso_now() -> String {
        let now: DateTime<Utc> = Utc::now();
        now.format("%Y-%m-%dT%H:%M:%S+00:00").to_string()
    }

    pub fn save(record: &mut SessionRecord) -> std::io::Result<PathBuf> {
        record.updated_at = iso_now();
        let path = session_path(&record.id);
        if let Some(p) = path.parent() { std::fs::create_dir_all(p)?; }
        let bytes = serde_json::to_vec_pretty(record)?;
        std::fs::write(&path, bytes)?;
        Ok(path)
    }

    /// Resolve by id prefix OR exact name match. Errors on ambiguity / not-found.
    pub fn load(id_or_name: &str) -> std::io::Result<SessionRecord> {
        let mut matches: Vec<SessionRecord> = Vec::new();
        let dir = state_dir();
        if !dir.exists() {
            return Err(std::io::Error::new(std::io::ErrorKind::NotFound,
                format!("no session: {}", id_or_name)));
        }
        for entry in std::fs::read_dir(&dir)? {
            let entry = entry?;
            if !entry.path().extension().map_or(false, |e| e == "json") { continue; }
            let bytes = std::fs::read(entry.path())?;
            let rec: SessionRecord = match serde_json::from_slice(&bytes) {
                Ok(r) => r, Err(_) => continue,
            };
            if rec.id.starts_with(id_or_name) || rec.name == id_or_name {
                matches.push(rec);
            }
        }
        match matches.len() {
            0 => Err(std::io::Error::new(std::io::ErrorKind::NotFound,
                format!("no session: {}", id_or_name))),
            1 => Ok(matches.into_iter().next().unwrap()),
            _ => {
                let names: Vec<String> = matches.iter().map(|r| r.id[..8].to_string()).collect();
                Err(std::io::Error::new(std::io::ErrorKind::InvalidInput,
                    format!("ambiguous session id; candidates: {}", names.join(", "))))
            }
        }
    }

    pub fn list_sessions() -> std::io::Result<Vec<SessionRecord>> {
        let dir = state_dir();
        if !dir.exists() { return Ok(Vec::new()); }
        let mut entries: Vec<(std::time::SystemTime, SessionRecord)> = Vec::new();
        for entry in std::fs::read_dir(&dir)? {
            let entry = entry?;
            if !entry.path().extension().map_or(false, |e| e == "json") { continue; }
            let mtime = entry.metadata().and_then(|m| m.modified())
                .unwrap_or(std::time::UNIX_EPOCH);
            let bytes = std::fs::read(entry.path())?;
            if let Ok(rec) = serde_json::from_slice::<SessionRecord>(&bytes) {
                entries.push((mtime, rec));
            }
        }
        // Newest first (matches Python sorted by mtime reverse=True).
        entries.sort_by(|a, b| b.0.cmp(&a.0));
        Ok(entries.into_iter().map(|(_, r)| r).collect())
    }

    pub fn delete(id: &str) -> std::io::Result<bool> {
        let path = session_path(id);
        if path.exists() { std::fs::remove_file(&path)?; Ok(true) } else { Ok(false) }
    }

    use voss_cli::session::{self, SessionRecord, Turn};

    /// Verify Rust→Python: Rust writes a session, Python loads it without error.
    #[test]
    fn rust_writes_python_reads() {
        let tmp = tempfile::tempdir().unwrap();
        std::env::set_var("XDG_STATE_HOME", tmp.path());
        std::env::set_var("HOME", tmp.path());

        let cwd = tmp.path().to_path_buf();
        let mut rec = SessionRecord::new(&cwd, "claude-sonnet-4-5", Some("rust-session"));
        rec.turns.push(Turn { role:"user".into(), content:"hello".into(), extra: Default::default() });
        rec.turns.push(Turn { role:"assistant".into(), content:"hi".into(), extra: Default::default() });
        let path = session::save(&mut rec).unwrap();
        assert!(path.exists());

        // Invoke Python: load the session, print fields. Skip if voss not importable.
        let out = Command::new("python3")
            .args(["-c", &format!(
                r#"import os; os.environ['XDG_STATE_HOME']={env!r}
from voss.harness import session as s
rec, _hist = s.load({sid!r})
print(rec.id, rec.name, len(rec.turns), rec.model)"#,
                env = tmp.path().to_string_lossy(), sid = rec.id)])
            .output();
        let out = match out { Ok(o) if o.status.success() => o, _ => {
            eprintln!("skipping rust→python parity (python voss unavailable)"); return;
        }};
        let s = String::from_utf8_lossy(&out.stdout);
        assert!(s.contains(&rec.id));
        assert!(s.contains("rust-session"));
        assert!(s.contains("2"), "expected 2 turns, output: {s}");
    }

    /// Verify Python→Rust: Python writes a session, Rust loads it.
    #[test]
    fn python_writes_rust_reads() {
        let tmp = tempfile::tempdir().unwrap();
        std::env::set_var("XDG_STATE_HOME", tmp.path());
        std::env::set_var("HOME", tmp.path());

        let out = Command::new("python3")
            .args(["-c", &format!(
                r#"import os; os.environ['XDG_STATE_HOME']={env!r}
from pathlib import Path
from voss.harness import session as s
from voss_runtime import EpisodicMemory
rec = s.SessionRecord.new(cwd=Path('.'), model='claude-sonnet-4-5', name='py-session')
mem = EpisodicMemory(capacity=40); mem.add('hi', role='user'); mem.add('ok', role='assistant')

        assert_eq!(rec.name, "py-session");
        assert_eq!(rec.turns.len(), 2);
        assert_eq!(rec.turns[0].role, "user");
