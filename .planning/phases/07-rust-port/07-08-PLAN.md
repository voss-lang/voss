    ```rust
    use async_trait::async_trait;
    use serde_json::{json, Value};
    use tokio::sync::Mutex;
    use std::sync::Arc;
    use uuid::Uuid;

    use crate::traits::{CompleteRequest, Message, ModelProvider, ProviderResponse};
    use voss_auth::CodexCreds;
    use voss_auth::{OPENAI_API_BASE, CHATGPT_BACKEND_BASE};

    /// Default model when caller doesn't specify (matches providers.py:218).
    const OPENAI_MODEL_DEFAULT: &str = "gpt-5";

    pub struct OpenAIOAuthProvider {
        creds: Arc<Mutex<CodexCreds>>,
        client: reqwest::Client,
        base_url: String,
        token_url_override: Option<String>,
        session_id: Uuid,
    }

    impl OpenAIOAuthProvider {
        pub fn new(creds: CodexCreds) -> Self {
            // ChatGPT-mode tokens go to chatgpt.com; api-key mode goes to api.openai.com.
            let base = if creds.auth_mode.eq_ignore_ascii_case("chatgpt") {
                CHATGPT_BACKEND_BASE
            } else {
                OPENAI_API_BASE
            };
            Self {
                creds: Arc::new(Mutex::new(creds)),
                client: reqwest::Client::builder()
                    .timeout(std::time::Duration::from_secs(120))
                    .build().expect("reqwest"),
                base_url: base.to_string(),
                token_url_override: None,
                session_id: Uuid::new_v4(),
            }
        }

        pub fn with_base_url(mut self, base: impl Into<String>) -> Self { self.base_url = base.into(); self }
        pub fn with_token_url(mut self, url: impl Into<String>) -> Self {
            self.token_url_override = Some(url.into()); self
        }

        async fn headers(&self) -> reqwest::header::HeaderMap {
            let creds = self.creds.lock().await;
            let mut h = reqwest::header::HeaderMap::new();
            if let Some(at) = &creds.access_token {
                h.insert("authorization", format!("Bearer {}", at).parse().unwrap());
            }
            h.insert("content-type", "application/json".parse().unwrap());
            h.insert("user-agent", "voss-harness/0.1".parse().unwrap());
            h.insert("originator", "codex_cli_rs".parse().unwrap());
            h.insert("OpenAI-Beta", "responses=v1".parse().unwrap());
            if let Some(aid) = &creds.account_id {
                h.insert("chatgpt-account-id", aid.parse().unwrap());
            }
            h
        }

        fn build_payload(&self, req: &CompleteRequest) -> Value {
            // Split system messages → top-level `instructions`; user/assistant → `input` items.
            let mut system_chunks: Vec<String> = Vec::new();
            let mut items: Vec<Value> = Vec::new();
            for m in &req.messages {
                match m.role.as_str() {
                    "system" => system_chunks.push(m.content.clone()),
                    role => {
                        let api_role = if role == "assistant" { "assistant" } else { "user" };
                        items.push(json!({
                            "type":"message",
                            "role": api_role,
                            "content": [{"type":"input_text","text": m.content}],
                        }));
                    }
                }
            }
            let mut body = json!({
                "model": if req.model.is_empty() { OPENAI_MODEL_DEFAULT.to_string() } else { req.model.clone() },
                "input": items,
                "store": false,
                "stream": false,
                "temperature": req.temperature,
            });
            if let Some(mt) = req.max_tokens { body["max_output_tokens"] = json!(mt); }
            if !system_chunks.is_empty() {
                body["instructions"] = Value::String(system_chunks.join("\n\n"));
            }
            if let Some(schema) = &req.response_schema {
                let mut schema = schema.clone();
                if let Some(obj) = schema.as_object_mut() { obj.insert("additionalProperties".into(), json!(false)); }
                let name = req.response_schema_name.as_deref().unwrap_or("Plan");
                body["text"] = json!({
                    "format": {
                        "type":"json_schema",
                        "name": name,
                        "strict": true,
                        "schema": schema,
                    }
                });
            }
            body
        }

        fn endpoint(&self) -> String {
            // chatgpt.com/backend-api/codex → /responses; api.openai.com → /v1/responses.
            if self.base_url.ends_with("/codex") {
                format!("{}/responses", self.base_url)
            } else {
                format!("{}/v1/responses", self.base_url)
            }
        }

        async fn force_refresh(&self) -> anyhow::Result<()> {
            let mut creds = self.creds.lock().await;
            voss_auth::refresh::refresh_codex(&mut *creds, &self.client,
                self.token_url_override.as_deref()).await?;
            Ok(())
        }
    }

    #[async_trait]
    impl ModelProvider for OpenAIOAuthProvider {
        async fn complete(&mut self, req: CompleteRequest) -> anyhow::Result<ProviderResponse> {
            let body = self.build_payload(&req);
            let url = self.endpoint();
            let post = || {
                let url = url.clone();
                let body = body.clone();
                async move {
                    let h = self.headers().await;
                    self.client.post(&url).headers(h).json(&body).send().await
                }
            };
            let mut resp = post().await?;
            if resp.status().as_u16() == 401 {
                let creds = self.creds.lock().await;
                let has_refresh = creds.refresh_token.is_some();
                drop(creds);
                if has_refresh {
                    self.force_refresh().await?;
                    resp = post().await?;
                }
            }
            let status = resp.status();
            if !status.is_success() {
                let txt = resp.text().await.unwrap_or_default();
                anyhow::bail!("OpenAI OAuth call failed [{}]: {}", status, &txt[..txt.len().min(500)]);
            }
            let data: Value = resp.json().await?;
            let usage = data.get("usage").cloned().unwrap_or(Value::Null);
            let prompt_tokens = usage.get("input_tokens").and_then(|v| v.as_u64()).unwrap_or(0) as u32;
            let completion_tokens = usage.get("output_tokens").and_then(|v| v.as_u64()).unwrap_or(0) as u32;

            let mut text = String::new();
            if let Some(output) = data.get("output").and_then(|v| v.as_array()) {
                for block in output {
                    if block.get("type").and_then(|v| v.as_str()) == Some("message") {
                        if let Some(content) = block.get("content").and_then(|v| v.as_array()) {
                            for c in content {
                                let ty = c.get("type").and_then(|v| v.as_str()).unwrap_or("");
                                if ty == "output_text" || ty == "text" {
                                    if let Some(t) = c.get("text").and_then(|v| v.as_str()) {
                                        text.push_str(t);
                                    }
                                }
                            }
                        }
                    }
                }
            }
            if text.is_empty() {
                if let Some(t) = data.get("output_text").and_then(|v| v.as_str()) { text = t.to_string(); }
            }

            let parsed = if req.response_schema.is_some() && !text.is_empty() {
                serde_json::from_str::<Value>(&text).ok()
            } else { None };

            Ok(ProviderResponse {
                text,
                model: data.get("model").and_then(|v| v.as_str()).unwrap_or(&req.model).to_string(),
                prompt_tokens, completion_tokens,
                cost_usd: 0.0,
                raw: data,
                parsed,
            })
        }
    }
    ```

    ```rust
    use std::path::PathBuf;
    use serde_json::Value;
    use voss_providers::{OpenAIOAuthProvider, CompleteRequest, Message, ModelProvider};
    use voss_auth::CodexCreds;
    use wiremock::{MockServer, Mock, ResponseTemplate};
    use wiremock::matchers::{method, path_regex};

    /// Fields whose values rotate per call. Compare keys exist; ignore values.
    const DYNAMIC_FIELDS: &[&str] = &["session_id", "request_id", "nonce"];

    fn fixtures_dir() -> PathBuf {
        let cargo = std::env::var("CARGO_MANIFEST_DIR").unwrap();
        PathBuf::from(cargo).parent().unwrap().parent().unwrap().join(".planning").join("codex-fixtures")
    }

    fn collect_fixtures() -> Vec<PathBuf> {
        let dir = fixtures_dir();
        if !dir.exists() { return Vec::new(); }
        std::fs::read_dir(&dir).unwrap()
            .filter_map(|e| e.ok())
            .filter(|e| e.path().extension().map_or(false, |x| x == "json"))
            .map(|e| e.path()).collect()
    }

    fn redact(v: &mut Value) {
        match v {
            Value::Object(m) => {
                for k in m.keys().cloned().collect::<Vec<_>>() {
                    if DYNAMIC_FIELDS.contains(&k.as_str()) {
                        m.insert(k, Value::String("<redacted>".into()));
                    } else {
                        redact(m.get_mut(&k).unwrap());
                    }
                }
            }
            Value::Array(a) => a.iter_mut().for_each(redact),
            _ => {}
        }
    }

    #[tokio::test]
    async fn replay_codex_fixtures() {
        let fixtures = collect_fixtures();
        if fixtures.is_empty() {
            panic!("no fixtures in .planning/codex-fixtures/ — Phase A of CODEX-OAUTH-PLAN.md must run first");
        }
        for fx in fixtures {
            let raw = std::fs::read(&fx).expect("read fixture");
            let fxv: Value = serde_json::from_slice(&raw).expect("parse fixture json");
            let captured_request_body = fxv.get("request").and_then(|r| r.get("body"))
                .cloned().expect("fixture has request.body");
            let captured_response_body = fxv.get("response").and_then(|r| r.get("body"))
                .cloned().expect("fixture has response.body");

            let server = MockServer::start().await;
            // Match either /responses (chatgpt) or /v1/responses (api).
            Mock::given(method("POST"))
                .and(path_regex(r"^(/v1)?/responses$"))
                .respond_with(ResponseTemplate::new(200).set_body_json(captured_response_body))
                .mount(&server).await;

            let creds = CodexCreds {
                api_key: None,
                access_token: Some("test-token".into()),
                refresh_token: Some("test-refresh".into()),
                account_id: Some("acct_test".into()),
                auth_mode: "chatgpt".into(),
            };
            let mut provider = OpenAIOAuthProvider::new(creds)
                .with_base_url(format!("{}/backend-api/codex", server.uri()));

            // Reconstruct CompleteRequest from captured request body.
            let model = captured_request_body.get("model").and_then(|v| v.as_str()).unwrap_or("gpt-5").to_string();
            let messages: Vec<Message> = captured_request_body.get("input").and_then(|v| v.as_array())
                .map(|a| a.iter().map(|item| {
                    let role = item.get("role").and_then(|v| v.as_str()).unwrap_or("user").to_string();
                    let content = item.get("content").and_then(|v| v.as_array())
                        .and_then(|a| a.first()).and_then(|c| c.get("text")).and_then(|v| v.as_str())
                        .unwrap_or("").to_string();
                    Message { role, content }
                }).collect()).unwrap_or_default();
            // Prepend instructions as a system message if present.
            let mut messages = messages;
            if let Some(inst) = captured_request_body.get("instructions").and_then(|v| v.as_str()) {
                messages.insert(0, Message { role: "system".into(), content: inst.into() });
            }

            let req = CompleteRequest {
                messages, model,
                temperature: captured_request_body.get("temperature").and_then(|v| v.as_f64()).unwrap_or(1.0) as f32,
                max_tokens: captured_request_body.get("max_output_tokens").and_then(|v| v.as_u64()).map(|v| v as u32),
                response_schema: captured_request_body.get("text").and_then(|t| t.get("format")).and_then(|f| f.get("schema")).cloned(),
                response_schema_name: captured_request_body.get("text").and_then(|t| t.get("format")).and_then(|f| f.get("name")).and_then(|v| v.as_str()).map(String::from),
                tools: None,
            };
            let _ = provider.complete(req).await;

            // Inspect what wiremock received.
            let received = server.received_requests().await.expect("requests")
                .into_iter().next().expect("at least one request");
            let actual_body: Value = serde_json::from_slice(&received.body).expect("actual body json");

            let mut a = actual_body.clone();
            let mut b = captured_request_body.clone();
            redact(&mut a); redact(&mut b);
            assert_eq!(a, b, "fixture {} request body mismatch.\nexpected: {}\nactual: {}",
                fx.display(),
                serde_json::to_string_pretty(&b).unwrap(),
                serde_json::to_string_pretty(&a).unwrap());
        }
    }
    ```
