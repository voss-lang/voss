    ```python
    """JSON-RPC over stdio server. Companion to crates/voss-bridge.
    LSP-style framing: Content-Length: <n>\\r\\n\\r\\n<json-body>.
    Versioned envelope: every response is {"v": 1, ...}.
    """
    from __future__ import annotations
    import json, sys
    from pathlib import Path

    PROTOCOL_VERSION = 1

    def _read_frame(stream) -> bytes | None:
        headers = {}
        while True:
            line = stream.readline()
            if not line:
                return None
            line = line.decode("ascii", errors="replace").rstrip("\\r\\n")
            if line == "":
                break
            if ":" in line:
                k, v = line.split(":", 1)
                headers[k.strip().lower()] = v.strip()
        n = int(headers.get("content-length", "-1"))
        if n < 0:
            raise ValueError("missing or negative Content-Length")
        return stream.read(n)

    def _write_frame(stream, body: bytes) -> None:
        stream.write(f"Content-Length: {len(body)}\\r\\n\\r\\n".encode("ascii"))
        stream.write(body)
        stream.flush()

    def _handle(req: dict) -> dict:
        method = req.get("method")
        params = req.get("params") or {}
        rid = req.get("id")
        try:
            if method == "ast":
                from voss.parser import parse
                from voss.ast_serializer import to_dict
                src = Path(params["path"]).read_text()
                program = parse(src, file=params["path"])
                result = {"v": PROTOCOL_VERSION, "program": to_dict(program)}
            elif method == "check":
                from voss.parser import parse
                from voss.analyzer import analyze
                src = Path(params["path"]).read_text()
                program = parse(src, file=params["path"])
                report = analyze(program)
                result = {"v": PROTOCOL_VERSION,
                          "diagnostics": [d.__dict__ for d in (report.errors + report.warnings)]}
            elif method == "compile":
                from voss.parser import parse
                from voss.analyzer import analyze
                from voss.codegen import generate_python
                src_path = Path(params["path"])
                program = parse(src_path.read_text(), file=str(src_path))
                analyze(program)
                py = generate_python(program)
                out = params.get("output") or str(src_path.with_suffix(".py"))
                Path(out).write_text(py)
                result = {"v": PROTOCOL_VERSION, "output": out, "ok": True}
            elif method == "run":
                # Compile then exec — stays a thin wrapper.
                result = {"v": PROTOCOL_VERSION, "ok": True, "note": "stub"}
            else:
                return {"jsonrpc": "2.0", "id": rid,
                        "error": {"code": -32601, "message": f"method not found: {method}"}}
        except Exception as e:
            return {"jsonrpc": "2.0", "id": rid,
                    "error": {"code": -32000, "message": str(e)}}
        return {"jsonrpc": "2.0", "id": rid, "result": result}

    def serve() -> None:
        stdin = sys.stdin.buffer
        stdout = sys.stdout.buffer
        while True:
            body = _read_frame(stdin)
            if body is None:
                return
            req = json.loads(body)
            resp = _handle(req)
            _write_frame(stdout, json.dumps(resp).encode("utf-8"))

    if __name__ == "__main__":
        serve()
    ```

    ```rust
    use std::path::{Path, PathBuf};
    use std::sync::atomic::{AtomicU64, Ordering};
    use tokio::io::{BufReader, AsyncWriteExt};
    use tokio::process::{Child, ChildStdin, ChildStdout, Command};
    use tokio::sync::Mutex;
    use serde_json::Value;
    use crate::framing::{read_frame, write_frame};

    pub struct PyBridge {
        python: PathBuf,
        child: Mutex<Option<BridgeChild>>,
        next_id: AtomicU64,
    }

    struct BridgeChild {
        _child: Child,
        stdin: ChildStdin,
        stdout: BufReader<ChildStdout>,
    }

    impl PyBridge {
        pub fn discover() -> std::io::Result<Self> {
            // Order: $VOSS_PYTHON, .venv/bin/python, python3 on PATH.
            let python = std::env::var_os("VOSS_PYTHON")
                .map(PathBuf::from)
                .or_else(|| {
                    let v = PathBuf::from(".venv/bin/python");
                    if v.exists() { Some(v) } else { None }
                })
                .unwrap_or_else(|| PathBuf::from("python3"));
            Ok(Self { python, child: Mutex::new(None), next_id: AtomicU64::new(1) })
        }

        async fn ensure_started(&self) -> std::io::Result<tokio::sync::MutexGuard<'_, Option<BridgeChild>>> {
            let mut guard = self.child.lock().await;
            if guard.is_none() {
                let mut child = Command::new(&self.python)
                    .args(["-m", "voss.bridge_server"])
                    .stdin(std::process::Stdio::piped())
                    .stdout(std::process::Stdio::piped())
                    .stderr(std::process::Stdio::inherit())
                    .spawn()?;
                let stdin = child.stdin.take().unwrap();
                let stdout = BufReader::new(child.stdout.take().unwrap());
                *guard = Some(BridgeChild { _child: child, stdin, stdout });
            }
            Ok(guard)
        }

        pub async fn call(&self, method: &str, params: Value) -> std::io::Result<Value> {
            let id = self.next_id.fetch_add(1, Ordering::SeqCst);
            let req = serde_json::json!({"jsonrpc":"2.0","id":id,"method":method,"params":params});
            let mut guard = self.ensure_started().await?;
            let bc = guard.as_mut().unwrap();
            let body = serde_json::to_vec(&req)?;
            write_frame(&mut bc.stdin, &body).await?;
            bc.stdin.flush().await?;
            let resp_bytes = read_frame(&mut bc.stdout).await?;
            let resp: Value = serde_json::from_slice(&resp_bytes)?;
            if let Some(err) = resp.get("error") {
                return Err(std::io::Error::new(std::io::ErrorKind::Other,
                    format!("bridge error: {}", err)));
            }
            Ok(resp.get("result").cloned().unwrap_or(Value::Null))
        }

        pub async fn ast(&self, path: &Path) -> std::io::Result<Value> {
            self.call("ast", serde_json::json!({"path": path.to_str().unwrap_or("")})).await
        }

        pub async fn check(&self, path: &Path) -> std::io::Result<Value> {
            self.call("check", serde_json::json!({"path": path.to_str().unwrap_or("")})).await
        }

        pub async fn compile(&self, path: &Path, output: Option<&Path>) -> std::io::Result<Value> {
            let params = if let Some(out) = output {
                serde_json::json!({"path": path.to_str().unwrap_or(""), "output": out.to_str().unwrap_or("")})
            } else {
                serde_json::json!({"path": path.to_str().unwrap_or("")})
            };
            self.call("compile", params).await
        }
    }
    ```
