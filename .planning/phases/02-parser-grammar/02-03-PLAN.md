# ----- Patterns (only inside match cases) -----
@dataclass(frozen=True, slots=True)
class SimilarPattern(Pattern):
    text: str

@dataclass(frozen=True, slots=True)
class WildcardPattern(Pattern):
    pass

@dataclass(frozen=True, slots=True)
class ExprPattern(Pattern):
    expr: Expr

# ----- Statements -----
@dataclass(frozen=True, slots=True)
class LetStmt(Stmt):
    name: str
    type_annot: TypeExpr | None = None
    value: Expr | None = None

@dataclass(frozen=True, slots=True)
class IfStmt(Stmt):
    # condition is either a regular Expr OR a ConfidenceGate (D-11).
    # Field type stays loose (Node) so the dataclass remains simple.
    condition: Node
    then_body: tuple[Stmt, ...]
    else_body: tuple[Stmt, ...] | None = None

@dataclass(frozen=True, slots=True)
class MatchCase(Node):
    pattern: Pattern
    body: tuple[Stmt, ...]

@dataclass(frozen=True, slots=True)
class MatchStmt(Stmt):
    scrutinee: Expr
    cases: tuple[MatchCase, ...]
    threshold: float | None = None    # set by @match_threshold (resolved question #2)

@dataclass(frozen=True, slots=True)
class CtxBlock(Stmt):
    budget: BudgetArg                  # the (single) budget kwarg in ctx(budget: 4000 tokens)
    body: tuple[Stmt, ...]

@dataclass(frozen=True, slots=True)
class WithinFallback(Stmt):
    budget_args: tuple[BudgetArg, ...]

@dataclass(frozen=True, slots=True)

@dataclass(frozen=True, slots=True)
class ReturnStmt(Stmt):
    value: Expr | None = None

@dataclass(frozen=True, slots=True)
class YieldStmt(Stmt):
    value: Expr | None = None

@dataclass(frozen=True, slots=True)

The executor MUST replace the existing `top_stmt: expr_stmt` line in the grammar file with `top_stmt: stmt`. Note the `match_threshold_stmt` rule eats the leading `@match_threshold(N)` and the following match together; the transformer lifts the value into the MatchStmt.

Important: `@match_threshold` must NOT collide with the general decorator rule that plan 02-04 adds for `@tool`. Disambiguation: `match_threshold_stmt` requires the literal IDENT `match_threshold` after `@`, so it lexes deterministically. With Earley + contextual lexer, both rules can coexist because plan 02-04's decorator rule will only match `@` followed by *non*-`match_threshold` identifiers. Document this in a comment in `grammar.lark`.

Actually, simpler approach: use a different non-terminal name and rely on Earley's preference for the longer match. To avoid ambiguity once plan 02-04 lands, keep `match_threshold_stmt` as the rule that captures `@match_threshold(N) <NEWLINE> match_stmt` as a single statement. Plan 02-04 will scope its decorator rule to declarations only (`@tool fn ...`, `@whatever class ...`), not to match statements. This keeps the two rules in distinct grammar slots.

Transformer methods:

```python
def stmt(self, meta, children):
    return children[0]

def let_stmt(self, meta, children):
    # children: [IDENT, type_expr?, expr?]   (Lark gives None for missing optionals with maybe_placeholders=True)
    name = str(children[0])
    type_annot = children[1] if len(children) > 1 else None
    value = children[2] if len(children) > 2 else None
    return LetStmt(span=_span(meta, self.file), name=name, type_annot=type_annot, value=value)

def if_condition(self, meta, children):
    return children[0]

def if_stmt(self, meta, children):
    # children: [if_condition, block(then), block(else)?]
    cond = children[0]
    then_body = children[1]
    else_body = children[2] if len(children) > 2 else None
    return IfStmt(span=_span(meta, self.file), condition=cond,
                  then_body=tuple(then_body), else_body=tuple(else_body) if else_body is not None else None)

def block(self, meta, children):
    # Drop NEWLINE tokens (children that are Lark Tokens of type NEWLINE).
    return tuple(c for c in children if not (hasattr(c, "type") and c.type == "NEWLINE"))

def match_stmt(self, meta, children):
    # children: [scrutinee, MatchCase, MatchCase, ...] (interspersed _NL stripped by underscore prefix? we use _NL with leading underscore so they're filtered)
    scrutinee = children[0]
    cases = tuple(c for c in children[1:] if isinstance(c, MatchCase))
    return MatchStmt(span=_span(meta, self.file), scrutinee=scrutinee, cases=cases, threshold=None)

def match_case(self, meta, children):
    pattern, body = children[0], children[1]
    body_tuple = body if isinstance(body, tuple) else (ExprStmt(span=_span(meta, self.file), expr=body),)
    return MatchCase(span=_span(meta, self.file), pattern=pattern, body=body_tuple)

def match_case_body(self, meta, children):
    c = children[0]
    if isinstance(c, tuple):  # came from `block` rule
        return c
    return c  # bare expr — match_case wraps it in ExprStmt

def pattern(self, meta, children):
    return children[0]

def similar_pattern(self, meta, children):
    raw = str(children[0])
    return SimilarPattern(span=_span(meta, self.file), text=raw[1:-1])

def wildcard_pattern(self, meta, children):
    return WildcardPattern(span=_span(meta, self.file))

def expr_pattern(self, meta, children):
    return ExprPattern(span=_span(meta, self.file), expr=children[0])

def match_threshold_stmt(self, meta, children):
    # children: [number_literal, MatchStmt]
    threshold_node = children[0]
    match_stmt = children[1]
    threshold = float(threshold_node.value)
    # Rebuild the MatchStmt with threshold set (frozen dataclass — use dataclasses.replace).
    from dataclasses import replace
    return replace(match_stmt, threshold=threshold)

def ctx_stmt(self, meta, children):
    budget = children[0]   # budget_kwarg → BudgetArg
    body = children[1]     # block
    return CtxBlock(span=_span(meta, self.file), budget=budget, body=tuple(body))

def within_stmt(self, meta, children):
    # children: [budget_kwarg+, block(primary), block(fallback)?]
    budget_args = []
    blocks = []
    for c in children:
        if isinstance(c, BudgetArg):
            budget_args.append(c)
        elif isinstance(c, tuple):
            blocks.append(c)
    primary = tuple(blocks[0]) if blocks else ()
    fallback = tuple(blocks[1]) if len(blocks) > 1 else None
    return WithinFallback(span=_span(meta, self.file), budget_args=tuple(budget_args),
                          primary=primary, fallback=fallback)

def budget_kwarg(self, meta, children):
    name = str(children[0])
    inner = children[1]   # BudgetArg from budget_literal — has unit/value/raw, name is ""
    return BudgetArg(span=_span(meta, self.file), name=name, unit=inner.unit, value=inner.value, raw=inner.raw)

def try_stmt(self, meta, children):
    # children: [block(try), IDENT?, block(catch)]
    try_body = children[0]
    if len(children) == 3:
        exc_name = str(children[1])
        catch_body = children[2]
    else:
        exc_name = None
        catch_body = children[1]
    return TryCatch(span=_span(meta, self.file),
                    try_body=tuple(try_body), exc_name=exc_name, catch_body=tuple(catch_body))

def return_stmt(self, meta, children):
    value = children[0] if children else None
    return ReturnStmt(span=_span(meta, self.file), value=value)

def yield_stmt(self, meta, children):
    value = children[0] if children else None
    return YieldStmt(span=_span(meta, self.file), value=value)

def include_stmt(self, meta, children):
    return IncludeStmt(span=_span(meta, self.file), value=children[0])
```

Note: with `maybe_placeholders=True` on the Lark parser, missing optional rule slots come through as `None` rather than being absent. Adjust the `len(children)` checks above accordingly — when a slot is `None`, treat it as missing. Concretely: in `let_stmt` and `if_stmt`, replace `len(children) > i` checks with `children[i] is not None` checks.
