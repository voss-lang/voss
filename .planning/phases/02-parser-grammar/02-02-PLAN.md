class Arg(Node):

class BinOp(Expr):
    op: str                # "+", "-", "*", "/", "==", "!=", "<", "<=", ">", ">=", "and", "or"
    left: Expr
    right: Expr

@dataclass(frozen=True, slots=True)
class UnaryOp(Expr):
    op: str                # "-", "not"
    operand: Expr

@dataclass(frozen=True, slots=True)
class Call(Expr):
    callee: Expr
    args: tuple[Arg, ...] = ()

@dataclass(frozen=True, slots=True)
class Member(Expr):
    obj: Expr
    attr: str

@dataclass(frozen=True, slots=True)
class Index(Expr):
    obj: Expr
    index: Expr

@dataclass(frozen=True, slots=True)
class ListLit(Expr):
    items: tuple[Expr, ...] = ()

@dataclass(frozen=True, slots=True)
class DictLit(Expr):
    # Each entry: (key, value). Key is Expr (typically StringLit or Identifier) per RESEARCH grammar `kv: (STRING|IDENT) ":" expr`.
    items: tuple[tuple[Expr, Expr], ...] = ()

@dataclass(frozen=True, slots=True)
class Param(Node):
    name: str
    type_annot: TypeExpr | None = None
    default: Expr | None = None

@dataclass(frozen=True, slots=True)
class Lambda(Expr):
    params: tuple[Param, ...]
    body: Expr

@dataclass(frozen=True, slots=True)
class SpawnExpr(Expr):
    agent: Call            # spawn always wraps a Call (e.g. `spawn Researcher(t)`)

@dataclass(frozen=True, slots=True)

    threshold: float

```
// ---- Type expressions (D-08, D-09) ----
type_expr: qual_name type_generics? type_kwargs?
qual_name: IDENT ("." IDENT)*
type_generics: "<" _NL* type_expr (_NL* "," _NL* type_expr)* _NL* ","? _NL* ">"
type_kwargs:   "(" _NL* type_kwarg (_NL* "," _NL* type_kwarg)* _NL* ","? _NL* ")"
type_kwarg:    IDENT ":" type_arg_value
type_arg_value: literal | budget_literal | qual_name

// ---- Expression precedence ladder (lowest -> highest) ----
expr:           lambda_expr | spawn_expr | or_expr
lambda_expr:    IDENT "=>" expr                                 -> lambda_single
              | "(" _NL* lambda_param_list? _NL* ")" "=>" expr  -> lambda_multi
lambda_param_list: lambda_param (_NL* "," _NL* lambda_param)* _NL* ","?
lambda_param:   IDENT (":" type_expr)?

spawn_expr:     "spawn" call_target

or_expr:        and_expr ("or" and_expr)*
and_expr:       not_expr ("and" not_expr)*
not_expr:       "not" not_expr                                  -> not_op
              | comparison
comparison:     additive (cmp_op additive)*
cmp_op:         "==" | "!=" | "<" | "<=" | ">" | ">="
additive:       multiplicative (add_op multiplicative)*
add_op:         "+" | "-"
multiplicative: unary (mul_op unary)*
mul_op:         "*" | "/"
unary:          "-" unary                                       -> neg_op
              | postfix

postfix:        primary (call_suffix | member_suffix | index_suffix)*
call_suffix:    "(" _NL* arg_list? _NL* ")"                     -> call_suf
member_suffix:  "." IDENT                                       -> member_suf
index_suffix:   "[" _NL* expr _NL* "]"                          -> index_suf

// `call_target` is what spawn wraps — must itself be a callable (postfix that ends in a call_suffix).
call_target:    postfix

primary:        literal
              | budget_literal
              | IDENT                                           -> ident_primary
              | "(" _NL* expr _NL* ")"                          -> paren_primary
              | list_lit
              | dict_lit

list_lit:       "[" _NL* (expr (_NL* "," _NL* expr)* _NL* ","?)? _NL* "]"
dict_lit:       "{" _NL* (kv (_NL* "," _NL* kv)* _NL* ","?)? _NL* "}"
kv:             dict_key ":" _NL* expr
dict_key:       STRING | IDENT

arg_list:       arg (_NL* "," _NL* arg)* _NL* ","?
arg:            named_arg | expr
named_arg:      IDENT ":" _NL* expr

// Confidence gate — exposed as a separate rule. Plan 02-03's `if` rule will reach this rule
// from the condition slot. It is NOT in the `expr` ladder (D-11).
confidence_gate: expr "@" "p" cmp_op number_literal
number_literal:  INT | FLOAT
```

Important notes:
- `_NL*` in argument/list/dict/generic positions enables continuation across newlines inside `(`, `[`, `<`, `{` (resolved open question #1 / RESEARCH risk #1).
- Trailing commas are accepted in `type_generics`, `type_kwargs`, `lambda_param_list`, `arg_list`, `list_lit`, `dict_lit` (resolved question #5).
- `spawn_expr: "spawn" call_target` keeps `spawn` left-prefix and prevents postfix chaining on the spawn result (RESEARCH note #4).
- `confidence_gate` appears here so plan 02-03 can reference it from `if` without re-introducing grammar.

# --- type_expr family ---
def qual_name(self, meta, children):
    parts = tuple(str(t) for t in children)
    return QualName(span=_span(meta, self.file), parts=parts)

def type_generics(self, meta, children):
    return tuple(c for c in children)  # already TypeRef nodes

def type_kwargs(self, meta, children):
    return tuple(c for c in children)  # already TypeKwarg nodes

def type_kwarg(self, meta, children):
    name = str(children[0])
    return TypeKwarg(span=_span(meta, self.file), name=name, value=children[1])

def type_arg_value(self, meta, children):
    return children[0]

def type_expr(self, meta, children):
    name = children[0]
    generics = ()
    kwargs = ()
    for c in children[1:]:
        if isinstance(c, tuple) and c and isinstance(c[0], TypeRef):
            generics = c
        elif isinstance(c, tuple) and c and isinstance(c[0], TypeKwarg):
            kwargs = c
        elif isinstance(c, tuple) and len(c) == 0:
            pass  # empty generics/kwargs
    return TypeRef(span=_span(meta, self.file), name=name, generics=generics, kwargs=kwargs)

# --- expression ladder ---
def or_expr(self, meta, children):
    return _left_assoc(self.file, meta, children, op_filter=lambda t: "or")

def and_expr(self, meta, children):
    return _left_assoc(self.file, meta, children, op_filter=lambda t: "and")

def not_op(self, meta, children):
    return UnaryOp(span=_span(meta, self.file), op="not", operand=children[0])

def comparison(self, meta, children):
    # children: [additive, cmp_op, additive, cmp_op, additive, ...]
    if len(children) == 1:
        return children[0]
    node = children[0]
    i = 1
    while i < len(children):
        op = str(children[i].children[0]) if hasattr(children[i], "children") else str(children[i])
        right = children[i + 1]
        node = BinOp(span=_span(meta, self.file), op=op, left=node, right=right)
        i += 2
    return node

def additive(self, meta, children):
    return _binop_chain(self.file, meta, children)

def multiplicative(self, meta, children):
    return _binop_chain(self.file, meta, children)

def neg_op(self, meta, children):
    return UnaryOp(span=_span(meta, self.file), op="-", operand=children[0])

def cmp_op(self, meta, children):
    # Return the operator string; comparison() looks at it.
    return str(children[0])

def add_op(self, meta, children):
    return str(children[0])

def mul_op(self, meta, children):
    return str(children[0])

def postfix(self, meta, children):
    node = children[0]
    for suf in children[1:]:
        kind, payload = suf  # tuple shape from suffix transformers
        if kind == "call":
            node = Call(span=_span(meta, self.file), callee=node, args=payload)
        elif kind == "member":
            node = Member(span=_span(meta, self.file), obj=node, attr=payload)
        elif kind == "index":
            node = Index(span=_span(meta, self.file), obj=node, index=payload)
    return node

def call_suf(self, meta, children):
    args = children[0] if children else ()
    if not isinstance(args, tuple):
        args = tuple(args)
    return ("call", args)

def member_suf(self, meta, children):
    return ("member", str(children[0]))

def index_suf(self, meta, children):
    return ("index", children[0])

def arg_list(self, meta, children):
    return tuple(children)

def named_arg(self, meta, children):
    return Arg(span=_span(meta, self.file), name=str(children[0]), value=children[1])

def arg(self, meta, children):
    c = children[0]
    if isinstance(c, Arg):
        return c
    return Arg(span=_span(meta, self.file), name=None, value=c)

def list_lit(self, meta, children):
    return ListLit(span=_span(meta, self.file), items=tuple(children))

def dict_lit(self, meta, children):
    return DictLit(span=_span(meta, self.file), items=tuple(children))

def kv(self, meta, children):
    return (children[0], children[1])

def dict_key(self, meta, children):
    c = children[0]
    if hasattr(c, "type") and c.type == "STRING":
        return StringLit(span=_span(meta, self.file), value=str(c)[1:-1], triple=False)
    return Identifier(span=_span(meta, self.file), name=str(c))

def ident_primary(self, meta, children):
    return Identifier(span=_span(meta, self.file), name=str(children[0]))

def paren_primary(self, meta, children):
    return children[0]

def primary(self, meta, children):
    return children[0]

def lambda_single(self, meta, children):
    name_tok, body = children
    p = Param(span=_span(meta, self.file), name=str(name_tok), type_annot=None, default=None)
    return Lambda(span=_span(meta, self.file), params=(p,), body=body)

def lambda_multi(self, meta, children):
    *params_or_empty, body = children
    params = params_or_empty[0] if params_or_empty and isinstance(params_or_empty[0], tuple) else ()
    return Lambda(span=_span(meta, self.file), params=tuple(params), body=body)

def lambda_param_list(self, meta, children):
    return tuple(children)

def lambda_param(self, meta, children):
    name = str(children[0])
    type_annot = children[1] if len(children) > 1 else None
    return Param(span=_span(meta, self.file), name=name, type_annot=type_annot, default=None)

def spawn_expr(self, meta, children):
    target = children[0]
    if not isinstance(target, Call):
        # spawn must wrap a Call — surface a parse error if grammar somehow yielded otherwise.
        raise VossParseError(file=self.file, line=meta.line, col=meta.column,
                             expected=["a function or agent call"], got=type(target).__name__)
    return SpawnExpr(span=_span(meta, self.file), agent=target)

def call_target(self, meta, children):
    return children[0]

def confidence_gate(self, meta, children):
    target, op, threshold_node = children[0], str(children[1]), children[2]
    threshold = float(threshold_node.value if hasattr(threshold_node, "value") else threshold_node)
    return ConfidenceGate(span=_span(meta, self.file), target=target, op=op, threshold=threshold)

def number_literal(self, meta, children):
    c = children[0]
    if hasattr(c, "type") and c.type == "FLOAT":
        return FloatLit(span=_span(meta, self.file), value=float(c))
    return IntLit(span=_span(meta, self.file), value=int(c))

def _binop_chain(file, meta, children):
    if len(children) == 1:
        return children[0]
    node = children[0]
    i = 1

        op = str(children[i])
